import pandas as pd

education_df = pd.read_csv("../../data/raw_data/Education2023.csv")

    col.strip()
    .lower()
    .replace(" ", "_")
    .replace(",", "")
    .replace('"', "")
    .replace("(", "")
    .replace(")", "")
    for col in education_df.columns
]

for col in education_df.columns:
    if education_df[col].dtype == "object":

    try:
        education_df[col] = pd.to_numeric(education_df[col], errors="ignore")

        continue  # This column contains non-numeric data, skip it

# Merge USDA Datasets
education = pd.read_csv("../../data/clean_data/education.csv")
poverty = pd.read_csv("../../data/clean_data/poverty.csv")
unemployment = pd.read_csv("../../data/clean_data/unemployment.csv")
population = pd.read_csv("../../data/clean_data/population.csv")


def smart_merge(left, right, on):
    overlapping_cols = set(left.columns).intersection(set(right.columns)) - {on}
    right_clean = right.drop(columns=overlapping_cols)
    return pd.merge(left, right_clean, on=on, how="outer")


merged_df = smart_merge(education, population, on="fips_code")
merged_df = smart_merge(merged_df, poverty, on="fips_code")
merged_df = smart_merge(merged_df, unemployment, on="fips_code")
